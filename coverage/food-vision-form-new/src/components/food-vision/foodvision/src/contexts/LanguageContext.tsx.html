import React, { createContext, useContext, useState, useEffect } from 'react';
import { useLocation } from 'react-router-dom';
 
type Language = 'he' | 'en';
 
type LanguageContextType = {
  language: Language;
  setLanguage: (lang: Language) => void;
  isRtl: boolean;
};
 
const defaultLanguage: Language = 'he';
 
const LanguageContext = createContext<LanguageContextType>({
  language: defaultLanguage,
  setLanguage: () => {},
  isRtl: true,
});
 
export const useLanguage = () => useContext(LanguageContext);
 
export const LanguageProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [language, setLanguageState] = useState<Language>(() => {
    // Try to get language from localStorage first
    const savedLang = localStorage.getItem('language');
    if (savedLang === 'he' || savedLang === 'en') {
      return savedLang;
    }
    return defaultLanguage;
  });
 
  const isRtl = language === 'he';
 
  // Safe wrapper for useLocation that won't crash outside Router context
  const safeLocation = () => {
    try {
      return useLocation();
    } catch (error) {
      console.info('Not in Router context, using default language:', defaultLanguage);
      return { pathname: '/' };
    }
  };
 
  // Use safe location or fallback
  const location = safeLocation();
 
  // Effect for path-based language detection
  useEffect(() => {
    try {
      if (location.pathname.startsWith('/en')) {
        setLanguageState('en');
      } else if (location.pathname === '/' || !location.pathname.includes('/en')) {
        // For root path or non-English paths, use Hebrew
        setLanguageState('he');
      }
    } catch (error) {
      console.info('Error in language effect:', error);
      // Default to Hebrew on error
      setLanguageState('he');
    }
  }, [location.pathname]);
 
  const setLanguage = (lang: Language) => {
    setLanguageState(lang);
    localStorage.setItem('language', lang);
    
    // Set document dir attribute for RTL/LTR support
    document.documentElement.dir = lang === 'he' ? 'rtl' : 'ltr';
    
    // Add/remove RTL class for global styling
    if (lang === 'he') {
      document.documentElement.classList.add('rtl');
    } else {
      document.documentElement.classList.remove('rtl');
    }
  };
 
  // Set initial document direction
  useEffect(() => {
    document.documentElement.dir = isRtl ? 'rtl' : 'ltr';
    if (isRtl) {
      document.documentElement.classList.add('rtl');
    } else {
      document.documentElement.classList.remove('rtl');
    }
  }, [isRtl]);
 
  return (
    <LanguageContext.Provider value={{ language, setLanguage, isRtl }}>
      {children}
    </LanguageContext.Provider>
  );
};
