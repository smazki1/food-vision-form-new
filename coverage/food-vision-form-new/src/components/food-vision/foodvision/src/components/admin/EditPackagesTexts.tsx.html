import React, { useEffect, useState } from "react";
import { supabase } from "@/integrations/supabase/client";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { toast } from "sonner";
import { Loader2 } from "lucide-react";
 
const LANGS = [
  { code: "he", label: "עברית" },
  { code: "en", label: "English" },
];
 
const PACKAGE_FIELDS = [
  // name/key, label
  ["package_vip_title", "VIP Package - Title"],
  ["package_vip_price", "VIP Package - Price"],
  ["package_vip_description", "VIP Package - Description"],
  ["package_vip_promo", "VIP Package - Promo"],

  ["package_vip_feature_5", "VIP Package - Feature 5"],
  ["package_vip_feature_6", "VIP Package - Feature 6"],
  ["package_vip_feature_7", "VIP Package - Feature 7"],
 
  ["package_full_title", "Full Package - Title"],
  ["package_full_price", "Full Package - Price"],
  ["package_full_old_price", "Full Package - Old Price"],

  ["additional_package_1_price", "Additional Package 1 - Price"],
  ["additional_package_1_details", "Additional Package 1 - Details"],

  ["additional_package_4_details", "Additional Package 4 - Details"],
];
 
type PackageTexts = Record<string, string>;
 
const EditPackagesTexts = () => {
  const [lang, setLang] = useState<"he" | "en">("he");
  const [data, setData] = useState<PackageTexts>({});
  const [loading, setLoading] = useState(false);
  const [saving, setSaving] = useState(false);
 
  useEffect(() => {
    setLoading(true);
    supabase
      .from("landing_texts")
      .select("name,value")
      .eq("lang", lang)
      .then(({ data, error }) => {
        setLoading(false);
        if (error) {
          toast.error("שגיאה בטעינת תוכן: " + error.message);
          return;
        }
        const dict: PackageTexts = {};
        data.forEach((row: { name: string; value: string }) => {
          dict[row.name] = row.value;
        });
        setData(dict);
      });
  }, [lang]);
 
  const handleChange = (name: string, value: string) => {
    setData((prev) => ({ ...prev, [name]: value }));
  };
 
  const handleSave = async () => {
    setSaving(true);
    const updates = PACKAGE_FIELDS.map(([key]) => ({
      name: key,
      value: data[key] || "",
      lang,
      updated_at: new Date().toISOString(),
    }));
 
    // UPSERT
    const { error } = await supabase.from("landing_texts")
      .upsert(updates, { onConflict: "name,lang" });
    setSaving(false);
    if (!error) {
      toast.success("הנתונים נשמרו בהצלחה (" + (lang === "he" ? "עברית" : "English") + ")");
    } else {
      toast.error("שגיאה בשמירה: " + error.message);
    }
  };
 
  return (
    <div className="p-4 max-w-3xl mx-auto">
      <Card>
        <CardHeader>
          <CardTitle>עריכת תוכן ומחירים – חבילות שירות</CardTitle>
          <div className="flex gap-3 mt-4">
            {LANGS.map(l => (
              <Button
                variant={lang === l.code ? "default" : "outline"}
                key={l.code}
                onClick={() => setLang(l.code as "he" | "en")}
                className={lang === l.code ? "bg-burgundy text-white" : ""}
              >
                {l.label}
              </Button>
            ))}
          </div>
        </CardHeader>
        <CardContent>
          {loading ? (
            <div className="text-center py-8"><Loader2 className="animate-spin mx-auto mb-2" /> טוען...</div>
          ) : (
            <form
              className="grid gap-5"
              onSubmit={e => {
                e.preventDefault();
                handleSave();
              }}
            >
              {PACKAGE_FIELDS.map(([key, label]) => (
                <div key={key}>
                  <label className="block font-bold mb-1">{label}:</label>
                  <input
                    type="text"
                    className="bg-gray-100 border px-3 py-2 rounded w-full"
                    value={data[key] || ""}
                    onChange={e => handleChange(key, e.target.value)}
                  />
                </div>
              ))}
              <Button
                type="submit"
                disabled={saving}
                className="mt-4 bg-burgundy text-white w-full"
              >
                {saving ? (
                  <><Loader2 className="animate-spin w-4 h-4 inline-block ml-2" />שומר...</>
                ) : "שמור את השינויים"}
              </Button>
            </form>
          )}
        </CardContent>
      </Card>
    </div>
  );
};
 
export default EditPackagesTexts;
