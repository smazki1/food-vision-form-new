import { useState, useEffect } from "react";
import { BeforeAfterPair } from "./gallery-data";
import { supabase } from "@/integrations/supabase/client";
 
interface UseGalleryStateProps {
  items: BeforeAfterPair[];
  itemsPerPage: number;
  useDatabaseImages?: boolean;
}
 
export const useGalleryState = ({ items, itemsPerPage, useDatabaseImages = false }: UseGalleryStateProps) => {
  const [hoverCardId, setHoverCardId] = useState<number | null>(null);
  const [isPaused, setIsPaused] = useState(false);
  const [activeImagePair, setActiveImagePair] = useState<number | null>(null);
  const [currentPage, setCurrentPage] = useState(0);
  const [showAfterImage, setShowAfterImage] = useState(true);
  const [galleryItems, setGalleryItems] = useState<BeforeAfterPair[]>(items);
  const [isLoading, setIsLoading] = useState(useDatabaseImages);
 
  // Fetch featured images from the database if useDatabaseImages is true
  useEffect(() => {
    if (useDatabaseImages) {
      const fetchFeaturedImages = async () => {
        try {
          setIsLoading(true);
          const { data, error } = await supabase
            .from('dishes')
            .select('*')
            .eq('is_featured', true)
            .not('before_url', 'is', null)
            .not('after_url', 'is', null)
            .order('display_order', { ascending: true });
            
          if (error) throw error;
          
          if (data && data.length > 0) {
            // Convert dishes data to BeforeAfterPair format
            const pairs: BeforeAfterPair[] = data.map((dish, index) => ({
              id: index,
              title: dish.title,
              before: dish.before_url || '',
              after: dish.after_url || '',
            }));
            
            setGalleryItems(pairs);
          }
        } catch (error) {
          console.error("Error fetching featured images:", error);
        } finally {
          setIsLoading(false);
        }
      };
      
      fetchFeaturedImages();
    }
  }, [useDatabaseImages]);
 
  const totalPages = Math.ceil(galleryItems.length / itemsPerPage);
  const startIndex = currentPage * itemsPerPage;
  const endIndex = startIndex + itemsPerPage;
  const currentItems = galleryItems.slice(startIndex, endIndex);
 
  const handleNextPage = () => {
    if (currentPage < totalPages - 1) {
      setCurrentPage(prev => prev + 1);
    }
  };
 
  const handlePrevPage = () => {
    if (currentPage > 0) {
      setCurrentPage(prev => prev - 1);
    }
  };
 
  const handleImageMouseEnter = (id: number) => {
    setHoverCardId(id);
  };
 
  const handleImageMouseLeave = () => {
    setHoverCardId(null);
  };
 
  const handleImageTouchStart = (id: number, currentId: number | null) => {
    // Handle touch interaction if needed
  };
 
  // Function to open full screen view
  const openFullScreenView = (index: number) => {
    setActiveImagePair(startIndex + index);
    setShowAfterImage(true); // Reset to "after" image when opening
  };
 
  // Function to close full screen view
  const closeFullScreenView = () => {
    setActiveImagePair(null);
    setShowAfterImage(true);
  };
 
  // Toggle between before and after images in mobile view
  const toggleBeforeAfter = () => {
    setShowAfterImage(prev => !prev);
  };
 
  return {
    hoverCardId,
    isPaused,
    activeImagePair,
    currentPage,
    showAfterImage,
    totalPages,
    startIndex,
    currentItems,
    galleryItems,
    isLoading,
    handleNextPage,
    handlePrevPage,
    handleImageMouseEnter,
    handleImageMouseLeave,
    handleImageTouchStart,
    openFullScreenView,
    closeFullScreenView,
    toggleBeforeAfter
  };
};
