import React from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Label } from "@/components/ui/label";
import { Input } from "@/components/ui/input";
import { Checkbox } from "@/components/ui/checkbox";
 
interface CarouselSettingsProps {
  isFeatured: boolean;
  setIsFeatured: (value: boolean) => void;
  displayOrder: number;
  setDisplayOrder: (value: number) => void;
}
 
const CarouselSettings: React.FC<CarouselSettingsProps> = ({
  isFeatured,
  setIsFeatured,
  displayOrder,
  setDisplayOrder,
}) => {
  return (
    <Card className="border-dashed border-2 border-burgundy/30 bg-burgundy/5">
      <CardHeader className="py-3">
        <CardTitle className="text-lg font-bold flex items-center gap-2">
          <span className="inline-block w-3 h-3 bg-burgundy rounded-full"></span>
          הגדרות לקרוסלה בדף הבית
        </CardTitle>
      </CardHeader>
      <CardContent className="py-3">
        <div className="flex items-center space-x-2 space-x-reverse mb-4">
          <Checkbox 
            id="isFeatured" 
            checked={isFeatured}
            onCheckedChange={(checked) => setIsFeatured(checked as boolean)}
          />
          <Label 
            htmlFor="isFeatured" 
            className="text-right cursor-pointer font-medium"
          >
            הצג תמונה זו בקרוסלה בדף הבית
          </Label>
        </div>
        
        {isFeatured && (
          <div className="space-y-2">
            <Label htmlFor="displayOrder" className="text-right block">סדר תצוגה בקרוסלה</Label>
            <div className="flex items-center gap-2">
              <Input
                id="displayOrder"
                type="number"
                min="0"
                step="1"
                value={displayOrder}
                onChange={(e) => setDisplayOrder(parseInt(e.target.value) || 0)}
                className="text-right w-32"
              />
              <span className="text-sm text-gray-500">(מספר נמוך יותר = מוצג ראשון)</span>
            </div>
            <p className="text-xs text-gray-500">* שים לב: תמונות יוצגו בקרוסלה רק אם הועלו גם תמונת "לפני" וגם תמונת "אחרי"</p>
          </div>
        )}
      </CardContent>
    </Card>
  );
};
 
export default CarouselSettings;
