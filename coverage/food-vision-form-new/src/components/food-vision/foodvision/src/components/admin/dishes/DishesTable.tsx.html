import React from "react";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Loader2, Star, AlertTriangle, Info, MoveVertical } from "lucide-react";
import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger } from "@/components/ui/tooltip";
import { type Dish } from "@/hooks/useDishesCRUD";
import DishActions from "./DishActions";
import { DragDropContext, Droppable, Draggable, DropResult } from "react-beautiful-dnd";
 
interface DishesTableProps {
  dishes: Dish[];
  isLoading: boolean;
  selectedDishes: Record<string, boolean>;
  onToggleSelection: (id: string) => void;
  onToggleSelectAll: () => void;
  onTogglePublish: (id: string, currentStatus: boolean) => void;
  onToggleFeatured?: (id: string, currentStatus: boolean) => void;
  onDeleteDish: (id: string) => void;
  onViewImage: (imageUrl: string) => void;
  onEditDish: (id: string) => void;
  onReorderDishes?: (result: DropResult) => void;
}
 
const DishesTable: React.FC<DishesTableProps> = ({
  dishes,
  isLoading,
  selectedDishes,
  onToggleSelection,
  onToggleSelectAll,
  onTogglePublish,
  onToggleFeatured,
  onDeleteDish,
  onViewImage,
  onEditDish,
  onReorderDishes
}) => {
  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-48">
        <Loader2 className="h-8 w-8 animate-spin text-burgundy" />
      </div>
    );
  }
  
  const handleDragEnd = (result: DropResult) => {
    if (!onReorderDishes || !result.destination) return;
    onReorderDishes(result);
  };
  
  return (
    <div className="bg-white rounded-lg shadow-sm overflow-hidden">
      <div className="overflow-x-auto">
        <DragDropContext onDragEnd={handleDragEnd}>
          <Table>
            <TableHeader>
              <TableRow>
                {onReorderDishes && (
                  <TableHead className="w-[30px]">
                    <span className="sr-only">סדר</span>
                  </TableHead>
                )}
                <TableHead className="w-[50px]">
                  <div className="flex items-center">
                    <input
                      type="checkbox"
                      checked={Object.keys(selectedDishes).length === dishes.length && dishes.length > 0}
                      onChange={onToggleSelectAll}
                      className="h-4 w-4 rounded border-gray-300 text-burgundy focus:ring-burgundy"
                    />
                  </div>
                </TableHead>
                <TableHead className="w-[100px]">תמונה</TableHead>
                <TableHead>כותרת</TableHead>
                <TableHead>תיאור</TableHead>
                <TableHead>תאריך העלאה</TableHead>
                <TableHead>סטטוס</TableHead>
                <TableHead>
                  <div className="flex items-center gap-1">
                    <Star className="h-4 w-4 fill-yellow-500 text-yellow-500" />
                    <span>קרוסלה</span>
                    <TooltipProvider>
                      <Tooltip>
                        <TooltipTrigger asChild>
                          <span className="inline-flex ml-1 cursor-help">
                            <Info className="h-4 w-4 text-gray-400" />
                          </span>
                        </TooltipTrigger>
                        <TooltipContent>
                          <p className="max-w-xs text-right">תמונות יוצגו בקרוסלה רק כאשר יש להם תמונת "לפני" ו"אחרי" והאפשרות "הצג בקרוסלה" מסומנת</p>
                        </TooltipContent>
                      </Tooltip>
                    </TooltipProvider>
                  </div>
                </TableHead>
                <TableHead>סדר</TableHead>
                <TableHead className="text-left">פעולות</TableHead>
              </TableRow>
            </TableHeader>
            <Droppable droppableId="dishes">
              {(provided) => (
                <TableBody
                  ref={provided.innerRef}
                  {...provided.droppableProps}
                >
                  {dishes.length > 0 ? (
                    dishes.map((dish, index) => {
                      const hasBothImages = !!dish.before_url && !!dish.after_url;
                      const isCarouselReady = dish.is_featured === true && hasBothImages;
                      
                      return (
                        <Draggable 
                          key={dish.id} 
                          draggableId={dish.id} 
                          index={index}
                          isDragDisabled={!onReorderDishes}
                        >
                          {(provided, snapshot) => (
                            <TableRow
                              ref={provided.innerRef}
                              {...provided.draggableProps}
                              className={snapshot.isDragging ? "bg-muted/30" : ""}
                            >
                              {onReorderDishes && (
                                <TableCell width={30}>
                                  <div 
                                    {...provided.dragHandleProps} 
                                    className="cursor-grab flex items-center justify-center"
                                  >
                                    <MoveVertical className="h-4 w-4 text-gray-400" />
                                  </div>
                                </TableCell>
                              )}
                              <TableCell>
                                <input
                                  type="checkbox"
                                  checked={!!selectedDishes[dish.id]}
                                  onChange={() => onToggleSelection(dish.id)}
                                  className="h-4 w-4 rounded border-gray-300 text-burgundy focus:ring-burgundy"
                                />
                              </TableCell>
                              <TableCell>
                                <img
                                  src={dish.img_thumb}
                                  alt={dish.title}
                                  className="h-12 w-12 rounded object-cover"
                                />
                              </TableCell>
                              <TableCell className="font-medium">{dish.title}</TableCell>
                              <TableCell>
                                {dish.caption === 'unpublished' ? '-' : (dish.caption || '-')}
                              </TableCell>
                              <TableCell>
                                {new Date(dish.created_at).toLocaleDateString('he-IL')}
                              </TableCell>
                              <TableCell>
                                <span
                                  className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${
                                    dish.is_published !== false
                                      ? 'bg-green-100 text-green-800'
                                      : 'bg-gray-100 text-gray-800'
                                  }`}
                                >
                                  {dish.is_published !== false ? 'מפורסם' : 'טיוטה'}
                                </span>
                              </TableCell>
                              <TableCell>
                                <div className="flex items-center gap-2">
                                  <span
                                    className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${
                                      isCarouselReady
                                        ? 'bg-yellow-100 text-yellow-800'
                                        : dish.is_featured === true && !hasBothImages
                                        ? 'bg-orange-100 text-orange-800'
                                        : 'bg-gray-100 text-gray-800'
                                    }`}
                                  >
                                    {isCarouselReady ? 'מוצג' : 
                                     dish.is_featured === true && !hasBothImages ? 'חסרות תמונות' : 'לא מוצג'}
                                  </span>
                                  
                                  {dish.is_featured === true && !hasBothImages && (
                                    <TooltipProvider>
                                      <Tooltip>
                                        <TooltipTrigger asChild>
                                          <span className="inline-flex cursor-help">
                                            <AlertTriangle className="h-4 w-4 text-orange-500" />
                                          </span>
                                        </TooltipTrigger>
                                        <TooltipContent>
                                          <p className="max-w-xs text-right">נבחר להופיע בקרוסלה אך חסרות תמונות "לפני" ו"אחרי". עריכת התמונה נדרשת.</p>
                                        </TooltipContent>
                                      </Tooltip>
                                    </TooltipProvider>
                                  )}
                                </div>
                              </TableCell>
                              <TableCell>
                                {isCarouselReady ? dish.display_order : '-'}
                              </TableCell>
                              <TableCell>
                                <DishActions
                                  dishId={dish.id}
                                  isPublished={dish.is_published !== false}
                                  isFeatured={dish.is_featured}
                                  hasBothImages={hasBothImages}
                                  onTogglePublish={onTogglePublish}
                                  onToggleFeatured={onToggleFeatured}
                                  onDelete={onDeleteDish}
                                  onView={onViewImage}
                                  onEdit={onEditDish}
                                  imageUrl={dish.img_full}
                                />
                              </TableCell>
                            </TableRow>
                          )}
                        </Draggable>
                      );
                    })
                  ) : (
                    <TableRow>
                      <TableCell colSpan={onReorderDishes ? 10 : 9} className="h-32 text-center text-gray-500">
                        אין תמונות להצגה
                      </TableCell>
                    </TableRow>
                  )}
                  {provided.placeholder}
                </TableBody>
              )}
            </Droppable>
          </Table>
        </DragDropContext>
      </div>
    </div>
  );
};
 
export default DishesTable;
