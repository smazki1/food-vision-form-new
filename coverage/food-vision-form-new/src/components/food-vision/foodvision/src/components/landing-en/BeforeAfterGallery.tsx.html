import { useState } from "react";
import { useIsMobile } from "@/hooks/use-mobile";
import { GalleryTitle } from "./gallery/GalleryTitle";
import { GalleryGrid } from "./gallery/GalleryGrid";
import { GalleryNavigation } from "./gallery/GalleryNavigation";
import { GalleryFullScreenView } from "./gallery/GalleryFullScreenView";
import { useGalleryState } from "./gallery/useGalleryState";
import { beforeAfterPairs } from "./gallery/gallery-data";
import { Loader2 } from "lucide-react";
 
const ITEMS_PER_PAGE = 2;
 
const BeforeAfterGallery = () => {
  const {
    hoverCardId,
    activeImagePair,
    currentPage,
    showAfterImage,
    totalPages,
    currentItems,
    galleryItems,
    isLoading,
    handleNextPage,
    handlePrevPage,
    handleImageMouseEnter,
    handleImageMouseLeave,
    handleImageTouchStart,
    openFullScreenView,
    closeFullScreenView,
    toggleBeforeAfter
  } = useGalleryState({ 
    items: beforeAfterPairs, // Fallback items
    itemsPerPage: ITEMS_PER_PAGE,
    useDatabaseImages: true // Set to true to fetch from the database
  });
 
  if (isLoading) {
    return (
      <section className="py-16 md:py-24 bg-gradient-to-b from-neutral-50 to-neutral-100 overflow-hidden">
        <div className="container mx-auto px-4 flex flex-col items-center justify-center min-h-[300px]">
          <Loader2 className="h-8 w-8 animate-spin text-burgundy mb-4" />
          <p className="text-gray-500">Loading images...</p>
        </div>
      </section>
    );
  }
  
  // If no items are available, don't render the component
  if (galleryItems.length === 0) {
    return null;
  }
 
  return (
    <section className="py-16 md:py-24 bg-gradient-to-b from-neutral-50 to-neutral-100 overflow-hidden">
      <div className="container mx-auto px-4">
        <GalleryTitle 
          title="From Regular Photos to Stunning Images - Without Changing the Actual Dish"
          description="Professional culinary photography that preserves all the authentic details of your dish – without expensive equipment or a professional photographer. Upload a simple phone photo and get a perfect result that's true to the original."
        />
        
        {/* Grid View */}
        <div className="relative">
          <GalleryGrid 
            items={currentItems}
            hoverCardId={hoverCardId}
            onImageMouseEnter={handleImageMouseEnter}
            onImageMouseLeave={handleImageMouseLeave}
            onImageTouchStart={handleImageTouchStart}
            onOpenFullScreen={openFullScreenView}
          />
 
          {/* Navigation Arrows */}
          {totalPages > 1 && (
            <GalleryNavigation 
              currentPage={currentPage}
              totalPages={totalPages}
              onPrevPage={handlePrevPage}
              onNextPage={handleNextPage}
            />
          )}
        </div>
      </div>
 
      {/* Full Screen View Dialog */}
      <GalleryFullScreenView 
        isOpen={activeImagePair !== null}
        onClose={closeFullScreenView}
        activeImagePair={activeImagePair}
        showAfterImage={showAfterImage}
        toggleBeforeAfter={toggleBeforeAfter}
        galleryItems={galleryItems}
      />
    </section>
  );
};
 
export default BeforeAfterGallery;
