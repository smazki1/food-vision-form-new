import React from "react";
import { ChevronLeft, ChevronRight } from "lucide-react";
import { BeforeAfterPair } from "@/components/landing/gallery/types";
 
interface GalleryFullScreenDesktopViewProps {
  currentItem: BeforeAfterPair;
  currentIndex: number;
  galleryItems: BeforeAfterPair[];
  goNext: () => void;
  goPrevious: () => void;
  setCurrentIndex: (index: number) => void;
}
 
export const GalleryFullScreenDesktopView: React.FC<GalleryFullScreenDesktopViewProps> = ({
  currentItem,
  currentIndex,
  galleryItems,
  goNext,
  goPrevious,
  setCurrentIndex
}) => {
  return (
    <>
      {/* Desktop Side by Side View */}
      <div className="w-full h-full flex flex-row">
        <div className="relative w-1/2 h-full flex items-center justify-center">
          <img
            src={currentItem.before}
            alt={`Before - ${currentItem.title}`}
            className="max-w-full max-h-full object-contain"
            draggable={false}
          />
          <div className="absolute top-4 left-4 px-3 py-1.5 rounded-full bg-black/70 text-white text-sm font-medium">
            Before
          </div>
        </div>
        <div className="relative w-1/2 h-full flex items-center justify-center">
          <img
            src={currentItem.after}
            alt={`After - ${currentItem.title}`}
            className="max-w-full max-h-full object-contain"
            draggable={false}
          />
          <div className="absolute top-4 left-4 px-3 py-1.5 rounded-full bg-orange text-white text-sm font-medium">
            After
          </div>
        </div>
      </div>
 
      {/* Desktop Navigation */}
      <button 
        onClick={goPrevious}
        className="absolute top-1/2 transform -translate-y-1/2 left-4 p-2 rounded-full bg-black/40 backdrop-blur-sm hover:bg-black/60 transition-all"
      >
        <ChevronLeft className="h-6 w-6 text-white" />
      </button>
      <button 
        onClick={goNext}
        className="absolute top-1/2 transform -translate-y-1/2 right-4 p-2 rounded-full bg-black/40 backdrop-blur-sm hover:bg-black/60 transition-all"
      >
        <ChevronRight className="h-6 w-6 text-white" />
      </button>
      
      {/* Desktop Indicator */}
      <div className="absolute bottom-4 left-1/2 transform -translate-x-1/2 px-4 py-2 rounded-full bg-black/40 backdrop-blur-sm">
        <div className="flex gap-1.5">
          {galleryItems.map((_, idx) => (
            <div 
              key={idx} 
              className={`w-2.5 h-2.5 rounded-full cursor-pointer ${idx === currentIndex ? 'bg-white' : 'bg-white/40'}`}
              onClick={() => setCurrentIndex(idx)}
            />
          ))}
        </div>
      </div>
    </>
  );
};
