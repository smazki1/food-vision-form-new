import React, { useState, useCallback } from "react";
import { Helmet } from "react-helmet";
import { useDropzone } from "react-dropzone";
import { supabase } from "@/integrations/supabase/client";
import { Upload as UploadIcon, X, Check, Loader2, AlertTriangle } from "lucide-react";
import { toast } from "sonner";
import { v4 as uuidv4 } from "uuid";
 
type UploadFile = {
  id: string;
  file: File;
  preview: string;
  progress: number;
  status: 'idle' | 'uploading' | 'success' | 'error';
  error?: string;
};
 
const UploadPage = () => {
  const [files, setFiles] = useState<UploadFile[]>([]);
  
  const onDrop = useCallback((acceptedFiles: File[]) => {
    const newFiles = acceptedFiles.slice(0, 50).map(file => ({
      id: uuidv4(),
      file,
      preview: URL.createObjectURL(file),
      progress: 0,
      status: 'idle' as const
    }));
    
    setFiles(prev => [...prev, ...newFiles].slice(0, 50));
  }, []);
  
  const { getRootProps, getInputProps, isDragActive } = useDropzone({
    onDrop,
    accept: {
      'image/jpeg': [],
      'image/png': [],
      'image/webp': []
    },
    maxFiles: 50,
  });
 
  const removeFile = (id: string) => {
    setFiles(files => {
      const updatedFiles = files.filter(file => file.id !== id);
      return updatedFiles;
    });
  };
 
  const uploadFiles = async () => {
    for (const fileObj of files) {
      if (fileObj.status === 'success') continue;
      
      try {
        // Update status to uploading
        setFiles(prev => prev.map(f => 
          f.id === fileObj.id ? { ...f, status: 'uploading', progress: 10 } : f
        ));
        
        // Generate a unique file name
        const fileName = `${fileObj.id}-${fileObj.file.name.replace(/[^a-zA-Z0-9.]/g, '_')}`;
        
        // Simulate processing progress
        setFiles(prev => prev.map(f => 
          f.id === fileObj.id ? { ...f, progress: 30 } : f
        ));
        
        // Upload to Supabase Storage
        const { error: uploadError } = await supabase.storage
          .from('dishes')
          .upload(`original/${fileName}`, fileObj.file);
          
        if (uploadError) throw uploadError;
        
        // In a real implementation, we'd call an edge function to generate thumbnails
        // For now, simulate this step
        setFiles(prev => prev.map(f => 
          f.id === fileObj.id ? { ...f, progress: 70 } : f
        ));
        
        // Create a record in the dishes table
        const bucketData = await supabase.storage.from('dishes').getPublicUrl(`original/${fileName}`);
        const imageUrl = bucketData.data.publicUrl;
        const { error: insertError } = await supabase
          .from('dishes')
          .insert({ 
            title: fileObj.file.name.split('.')[0].replace(/_/g, ' '),
            img_thumb: imageUrl, // In a real app, this would be a thumbnail URL
            img_full: imageUrl
          });
          
        if (insertError) throw insertError;
        
        // Update status to success
        setFiles(prev => prev.map(f => 
          f.id === fileObj.id ? { ...f, status: 'success', progress: 100 } : f
        ));
        
      } catch (error) {
        console.error("Error uploading file:", error);
        setFiles(prev => prev.map(f => 
          f.id === fileObj.id ? { ...f, status: 'error', error: 'שגיאה בהעלאה' } : f
        ));
      }
    }
    
    toast.success('התמונות הועלו בהצלחה!');
  };
 
  const allFilesProcessed = files.length > 0 && files.every(f => f.status === 'success' || f.status === 'error');
  const someFilesReady = files.length > 0 && files.some(f => f.status === 'idle');
  
  return (
    <>
      <Helmet>
        <title>העלאת תמונות - Food Vision AI</title>
      </Helmet>
      
      <div className="p-6 max-w-7xl mx-auto">
        <div className="mb-8">
          <h1 className="text-2xl font-bold text-gray-800">העלאת תמונות</h1>
          <p className="text-gray-600">גרור תמונות לתוך האזור המסומן או לחץ לבחירת קבצים.</p>
        </div>
        
        <div 
          {...getRootProps()} 
          className={`border-2 border-dashed rounded-lg p-10 text-center cursor-pointer transition-colors ${
            isDragActive ? 'border-burgundy bg-burgundy/5' : 'border-gray-300 hover:border-burgundy/60'
          }`}
        >
          <input {...getInputProps()} />
          <UploadIcon className="h-12 w-12 mx-auto text-gray-400" />
          <p className="mt-4 text-gray-500">גרור תמונות לכאן או לחץ לבחירת קבצים</p>
          <p className="text-sm text-gray-400 mt-2">ניתן להעלות עד 50 תמונות בפורמט JPG, PNG או WEBP</p>
        </div>
 
        {files.length > 0 && (
          <div className="mt-8">
            <div className="flex items-center justify-between mb-4">
              <h2 className="text-lg font-semibold">התמונות שלך ({files.length})</h2>
              <div className="space-x-2 ml-2">
                <button
                  onClick={() => setFiles([])}
                  className="px-3 py-1 text-sm border border-gray-300 rounded-md hover:bg-gray-50 transition-colors"
                >
                  נקה הכל
                </button>
                {someFilesReady && (
                  <button
                    onClick={uploadFiles}
                    disabled={!someFilesReady}
                    className="px-3 py-1 text-sm bg-burgundy text-white rounded-md hover:bg-burgundy/90 transition-colors disabled:opacity-50"
                  >
                    העלה תמונות
                  </button>
                )}
                {allFilesProcessed && (
                  <button
                    onClick={() => setFiles([])}
                    className="px-3 py-1 text-sm bg-green-600 text-white rounded-md hover:bg-green-700 transition-colors"
                  >
                    העלאה הסתיימה
                  </button>
                )}
              </div>
            </div>
            
            <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
              {files.map((fileObj) => (
                <div key={fileObj.id} className="relative bg-white rounded-lg shadow-sm overflow-hidden group">
                  <img
                    src={fileObj.preview}
                    alt={fileObj.file.name}
                    className="w-full aspect-square object-cover"
                    onLoad={() => URL.revokeObjectURL(fileObj.preview)}
                  />
                  
                  {/* File status overlay */}
                  {fileObj.status !== 'idle' && (
                    <div className="absolute inset-0 bg-black/60 flex flex-col items-center justify-center text-white">
                      {fileObj.status === 'uploading' && (
                        <>
                          <Loader2 className="h-10 w-10 animate-spin mb-2" />
                          <p className="text-sm font-medium">{fileObj.progress}%</p>
                        </>
                      )}
                      {fileObj.status === 'success' && (
                        <>
                          <Check className="h-10 w-10 text-green-400 mb-2" />
                          <p className="text-sm font-medium">הועלה בהצלחה</p>
                        </>
                      )}
                      {fileObj.status === 'error' && (
                        <>
                          <AlertTriangle className="h-10 w-10 text-red-400 mb-2" />
                          <p className="text-sm font-medium">{fileObj.error || 'שגיאה בהעלאה'}</p>
                        </>
                      )}
                    </div>
                  )}
                  
                  {/* Remove button */}
                  <button
                    onClick={() => removeFile(fileObj.id)}
                    className="absolute top-2 left-2 bg-black/70 text-white p-1 rounded-full opacity-0 group-hover:opacity-100 transition-opacity"
                    aria-label="הסר"
                  >
                    <X size={16} />
                  </button>
                  
                  <div className="p-2">
                    <p className="truncate text-sm">
                      {fileObj.file.name}
                    </p>
                    <p className="text-xs text-gray-500">
                      {(fileObj.file.size / 1024 / 1024).toFixed(2)} MB
                    </p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    </>
  );
};
 
export default UploadPage;
