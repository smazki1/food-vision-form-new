import React, { useState, useEffect } from "react";
import { Helmet } from "react-helmet";
import { supabase } from "@/integrations/supabase/client";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Loader2, Edit, Trash2, Eye, Check } from "lucide-react";
import { toast } from "sonner";
 
type Dish = {
  id: string;
  title: string;
  caption: string | null;
  img_thumb: string;
  created_at: string;
  is_published?: boolean;
};
 
const Manage = () => {
  const [dishes, setDishes] = useState<Dish[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [selectedDishes, setSelectedDishes] = useState<Record<string, boolean>>({});
  
  useEffect(() => {
    const fetchDishes = async () => {
      try {
        const { data, error } = await supabase
          .from('dishes')
          .select('*')
          .order('created_at', { ascending: false });
          
        if (error) throw error;
        
        setDishes(data || []);
      } catch (error) {
        console.error("Error fetching dishes:", error);
        toast.error("שגיאה בטעינת התמונות");
      } finally {
        setIsLoading(false);
      }
    };
    
    fetchDishes();
  }, []);
  
  const toggleSelection = (id: string) => {
    setSelectedDishes(prev => ({
      ...prev,
      [id]: !prev[id]
    }));
  };
  
  const toggleSelectAll = () => {
    if (Object.keys(selectedDishes).length === dishes.length) {
      // Deselect all
      setSelectedDishes({});
    } else {
      // Select all
      const newSelected: Record<string, boolean> = {};
      dishes.forEach(dish => {
        newSelected[dish.id] = true;
      });
      setSelectedDishes(newSelected);
    }
  };
  
  const togglePublish = async (id: string, currentStatus: boolean) => {
    try {
      // Optimistically update UI
      setDishes(prev => prev.map(dish => 
        dish.id === id ? { ...dish, is_published: !currentStatus } : dish
      ));
      
      // In a real implementation, we'd update the is_published field in the database
      toast.success(`התמונה ${!currentStatus ? 'פורסמה' : 'הוסרה מהפרסום'} בהצלחה`);
    } catch (error) {
      console.error("Error toggling publish status:", error);
      toast.error("שגיאה בעדכון סטטוס הפרסום");
      
      // Revert optimistic update
      setDishes(prev => prev.map(dish => 
        dish.id === id ? { ...dish, is_published: currentStatus } : dish
      ));
    }
  };
  
  const deleteDish = async (id: string) => {
    if (!confirm("האם אתה בטוח שברצונך למחוק תמונה זו?")) return;
    
    try {
      // Optimistically update UI
      setDishes(prev => prev.filter(dish => dish.id !== id));
      
      // In a real implementation, we'd delete from the database and storage
      toast.success("התמונה נמחקה בהצלחה");
    } catch (error) {
      console.error("Error deleting dish:", error);
      toast.error("שגיאה במחיקת התמונה");
      
      // Refetch dishes to undo optimistic update
      const { data } = await supabase
        .from('dishes')
        .select('*')
        .order('created_at', { ascending: false });
        
      setDishes(data || []);
    }
  };
  
  const getSelectedCount = () => {
    return Object.values(selectedDishes).filter(Boolean).length;
  };
  
  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-[calc(100vh-64px)]">
        <Loader2 className="h-8 w-8 animate-spin text-burgundy" />
      </div>
    );
  }
  
  return (
    <>
      <Helmet>
        <title>ניהול תמונות - Food Vision AI</title>
      </Helmet>
      
      <div className="p-6 max-w-7xl mx-auto">
        <div className="flex justify-between items-center mb-8">
          <div>
            <h1 className="text-2xl font-bold text-gray-800">ניהול תמונות</h1>
            <p className="text-gray-600">ניהול ועריכת התמונות בקטלוג</p>
          </div>
          
          {getSelectedCount() > 0 && (
            <div className="flex items-center gap-2">
              <span className="text-sm text-gray-500">
                {getSelectedCount()} תמונות נבחרו
              </span>
              <button 
                className="px-3 py-1 bg-red-600 text-white rounded-md text-sm hover:bg-red-700 transition-colors"
                onClick={() => {
                  toast("פעולה זו עדיין לא זמינה", {
                    description: "בקרוב יתווספו פעולות עריכה מרובות"
                  });
                }}
              >
                מחק נבחרים
              </button>
            </div>
          )}
        </div>
        
        <div className="bg-white rounded-lg shadow-sm overflow-hidden">
          <div className="overflow-x-auto">
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead className="w-[50px]">
                    <div className="flex items-center">
                      <input
                        type="checkbox"
                        checked={Object.keys(selectedDishes).length === dishes.length && dishes.length > 0}
                        onChange={toggleSelectAll}
                        className="h-4 w-4 rounded border-gray-300 text-burgundy focus:ring-burgundy"
                      />
                    </div>
                  </TableHead>
                  <TableHead className="w-[100px]">תמונה</TableHead>
                  <TableHead>כותרת</TableHead>
                  <TableHead>תיאור</TableHead>
                  <TableHead>תאריך העלאה</TableHead>
                  <TableHead>סטטוס</TableHead>
                  <TableHead className="text-left">פעולות</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {dishes.map((dish) => (
                  <TableRow key={dish.id}>
                    <TableCell>
                      <input
                        type="checkbox"
                        checked={!!selectedDishes[dish.id]}
                        onChange={() => toggleSelection(dish.id)}
                        className="h-4 w-4 rounded border-gray-300 text-burgundy focus:ring-burgundy"
                      />
                    </TableCell>
                    <TableCell>
                      <img
                        src={dish.img_thumb}
                        alt={dish.title}
                        className="h-12 w-12 rounded object-cover"
                      />
                    </TableCell>
                    <TableCell className="font-medium">{dish.title}</TableCell>
                    <TableCell>{dish.caption || '-'}</TableCell>
                    <TableCell>
                      {new Date(dish.created_at).toLocaleDateString('he-IL')}
                    </TableCell>
                    <TableCell>
                      <span
                        className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${
                          dish.is_published !== false
                            ? 'bg-green-100 text-green-800'
                            : 'bg-gray-100 text-gray-800'
                        }`}
                      >
                        {dish.is_published !== false ? 'מפורסם' : 'טיוטה'}
                      </span>
                    </TableCell>
                    <TableCell>
                      <div className="flex items-center gap-2">
                        <button
                          onClick={() => togglePublish(dish.id, dish.is_published !== false)}
                          className={`p-1 rounded-full ${
                            dish.is_published !== false
                              ? 'text-gray-500 hover:bg-gray-100'
                              : 'text-green-600 hover:bg-green-50'
                          } transition-colors`}
                          title={dish.is_published !== false ? 'הסר מפרסום' : 'פרסם'}
                        >
                          <Check className="h-4 w-4" />
                        </button>
                        <button
                          onClick={() => toast.info("עמוד עריכה בפיתוח")}
                          className="p-1 rounded-full text-blue-600 hover:bg-blue-50 transition-colors"
                          title="ערוך"
                        >
                          <Edit className="h-4 w-4" />
                        </button>
                        <button
                          onClick={() => deleteDish(dish.id)}
                          className="p-1 rounded-full text-red-600 hover:bg-red-50 transition-colors"
                          title="מחק"
                        >
                          <Trash2 className="h-4 w-4" />
                        </button>
                        <button
                          onClick={() => toast.info("תצוגה מקדימה בפיתוח")}
                          className="p-1 rounded-full text-gray-500 hover:bg-gray-100 transition-colors"
                          title="תצוגה מקדימה"
                        >
                          <Eye className="h-4 w-4" />
                        </button>
                      </div>
                    </TableCell>
                  </TableRow>
                ))}
                
                {dishes.length === 0 && (
                  <TableRow>
                    <TableCell colSpan={7} className="h-32 text-center text-gray-500">
                      אין תמונות להצגה
                    </TableCell>
                  </TableRow>
                )}
              </TableBody>
            </Table>
          </div>
        </div>
      </div>
    </>
  );
};
 
export default Manage;
